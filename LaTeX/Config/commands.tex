% If/Else to customize final report and chapters separately
\def\docctrl{chapter}
\newcommand{\case}[2]{
	\ifx\doc\docctrl
		#1
	\else
		#2
	\fi
}

% Centred figure command
\newcommand{\cfig}[3]{
\begin{figure}[H]
	\centering
	\includegraphics[width=#2\linewidth]{Images/#1}
    \caption{#3}
	\label{fig:#1}
\end{figure}
}

% Figure on right side
\newcommand{\rfig}[3]{
\begin{wrapfigure}{r}{#3\linewidth}
	\centering
	\includegraphics[width=\linewidth]{Images/#1}
	\caption{#2}
	\label{fig:#1}
	\vspace*{-13pt}
\end{wrapfigure}
}

\newcommand{\rfigII}[6]{
\begin{wrapfigure}[#4]{r}{#3\linewidth}
	\vspace*{#5}
	\centering
	\includegraphics[width=\linewidth]{Images/#1}
	\vspace*{#6}
	\caption{#2}
	\label{fig:#1}
\end{wrapfigure}
}

% Two figures side by side
\newcommand{\twofig}[5]{
\begin{figure}[H]
	\begin{minipage}{0.48\linewidth}
		\centering
		\includegraphics[width=#5\linewidth]{Images/#1}
		\caption{#2}
		\label{fig:#1}
	\end{minipage}\hfill
	\begin{minipage}{0.48\linewidth}
		\centering
		\includegraphics[width=#5\linewidth]{Images/#3}
		\caption{#4}
		\label{fig:#3}
	\end{minipage}
\end{figure}
}

% Create new chapter (with sub-bibliography)
\newcommand{\newchapter}[1]{
\setcounter{chapter}{#1}
\setcounter{section}{0}
\setcounter{figure}{0}
\setcounter{table}{0}
\setcounter{equation}{0}
\chapter*{Homework #1}
\addcontentsline{toc}{chapter}{Homework #1}
\begin{refsection}
	\import{Chapters/H#1/}{Body}
	\pagebreak
	\phantomsection
	\printbibliography[heading=subbibintoc,title=Bibliography]
\end{refsection}
}

% Create first page
\newcommand{\firstpage}[1]{
	\def\tit{#1}
	\case{\import{../../Config/}{title_page_chapter}}{\import{Config/}{title_page_chapter}}
}

% Create body and bibliography of the chapter
\newcommand{\bodyandbib}{
\begin{refsection}
	\import{}{Body}
	\pagebreak
	\phantomsection
	\printbibliography[heading=bibintoc,title=Bibliography]
\end{refsection}
}
