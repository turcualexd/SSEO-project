% Centred figure command
\newcommand{\cfig}[3]{
\begin{figure}[H]
	\centering
	\includegraphics[width=#2\linewidth]{Images/#1}
    \caption{#3}
	\label{fig:#1}
\end{figure}
}

% Create new chapter (with sub-bibliography)
\newcommand{\newchapter}[2]{
\chapter{#2}
\begin{refsection}
	\import{Chapters/H#1/}{Body}
	\printbibliography[heading=subbibintoc,title=Bibliography]
\end{refsection}
}

% Create body and bibliography of the chapter
\newcommand{\bodyandbib}{
\begin{refsection}
	\import{}{Body}
	\printbibliography[heading=bibintoc,title=Bibliography]
\end{refsection}
}
